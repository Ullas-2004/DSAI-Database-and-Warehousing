CREATE TABLE `meta_lineage_edge` (
  `edge_id` varchar(64) NOT NULL,
  `from_table` varchar(100) NOT NULL,
  `from_key` varchar(100) NOT NULL,
  `to_table` varchar(100) NOT NULL,
